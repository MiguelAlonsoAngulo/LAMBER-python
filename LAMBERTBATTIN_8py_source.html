6#------------------------------------------------------------------------------
7#
8# LAMBERBATTIN
9#
10# this subroutine solves Lambert's problem using Battins method. The method
11# is developed in Battin (1987). It uses contiNued fractions to speed the
12# solution and has several parameters that are defined differently than
13# the traditional Gaussian technique.
14#
15# Inputs: Description Range/Units
16# ro - IJK Position vector 1 m
17# r - IJK Position vector 2 m
18# dm - direction of motion 'pro','retro'
19# Dtsec - Time between ro and r s
20#
21# OutPuts:
22# vo - IJK Velocity vector m/s
23# v - IJK Velocity vector m/s
24#
25# Reference:
26# Vallado D. A; Fundamentals of Astrodynamics and Applications; McGraw-Hill
27# , New York; 3rd edition(2007).
28#
29# Last modified: 2015/08/12 M. Mahooti
30#
31#------------------------------------------------------------------------------
32def LAMBERTBATTIN(ro, r, dm, Dtsec):
33 small = 0.000001;
34 mu = 3.986004418e14; # m3/s2
35 y1 = 0;
36 magr = v_norm(r);
37 magro = v_norm(ro);
38 CosDeltaNu= v_dot(ro,r)/(magro*magr);
39 rcrossr = v_cross(ro,r);
40 magrcrossr = v_norm(rcrossr);
41 if (dm == 'pro'):
42 SinDeltaNu = magrcrossr/(magro*magr);
43 else:
44 SinDeltaNu = -magrcrossr/(magro*magr);
45
46 DNu = atan2(SinDeltaNu,CosDeltaNu);
47
48 # the angle needs to be positive to work for the long way
49 if (DNu < 0.0):
50 DNu = 2.0*pi+DNu;
51
52 RoR = magr/magro;
53 eps = RoR - 1.0;
54 tan2w = 0.25*eps*eps / ( sqrt( RoR ) + RoR * ( 2.0 + sqrt( RoR ) ) );
55 rp = sqrt( magro*magr )*( (cos(DNu*0.25))**2 + tan2w );
56 if ( DNu < pi ):
57 L = ( (sin(DNu*0.25))**2 + tan2w ) / ( (sin(DNu*0.25))**2 + tan2w + cos( DNu*0.5 ) );
58 else:
59 L = ( (cos(DNu*0.25))**2 + tan2w - cos( DNu*0.5 ) ) / ( (cos(DNu*0.25))**2 + tan2w );
60
61 m = mu*Dtsec*Dtsec / ( 8.0*rp*rp*rp );
62 x = 10.0;
63 xn = L;
64 chord= sqrt( magro*magro + magr*magr - 2.0*magro*magr*cos( DNu ) );
65 s = ( magro + magr + chord )*0.5;
66 lim1 = sqrt(m/L);
67
68 Loops = 1;
69 while (True):
70 x = xn;
71 tempx= seebatt(x);
72 Denom= 1.0 / ( (1.0+2.0*x+L) * (4.0*x + tempx*(3.0+x) ) );
73 h1 = ( L+x )**2 * ( 1.0+ 3.0*x + tempx )*Denom;
74 h2 = m*( x - L + tempx )*Denom;
75
76 # ----------------------- Evaluate CUBIC ------------------
77 b = 0.25*27.0*h2 / ((1.0+h1)**3 );
78 if (b < -1.0): # reset the initial condition
79 xn = 1.0 - 2.0*l;
80 else:
81 if (y1 > lim1):
82 xn = xn * (lim1/y1);
83 else:
84 u = 0.5*b / ( 1.0 + sqrt( 1.0 + b ) );
85 k2 = seebattk(u);
86 y = ( ( 1.0+h1 ) / 3.0 )*( 2.0 + sqrt( 1.0+b )/( 1.0+2.0*u*k2*k2 ) );
87 xn= sqrt( ( (1.0-L)*0.5 )**2 + m/(y*y) ) - ( 1.0+L )*0.5;
88
89 Loops = Loops + 1;
90 y1 = sqrt(m/((L+x)*(1.0+x)) );
91 if ((abs(xn-x) < small) and (Loops > 30)):
92 break;
93
94 a = mu*Dtsec*Dtsec / (16.0*rp*rp*xn*y*y );
95
96 # ------------------ Find Eccentric anomalies -----------------
97 # ------------------------ Hyperbolic -------------------------
98 if ( a < -small ):
99 arg1 = sqrt( s / ( -2.0*a ) );
100 arg2 = sqrt( ( s-chord ) / ( -2.0*a ) );
101 # ------- Evaluate f and g functions --------
102 AlpH = 2.0 * asinh( arg1 );
103 BetH = 2.0 * asinh( arg2 );
104 DH = AlpH - BetH;
105 F = 1.0 - (a/magro)*(1.0 - cosh(DH) );
106 GDot = 1.0 - (a/magr) *(1.0 - cosh(DH) );
107 G = Dtsec - sqrt(-a*a*a/mu)*(sinh(DH)-DH);
108 else:
109 # ------------------------ Elliptical ---------------------
110 if ( a > small ):
111 arg1 = sqrt( s / ( 2.0*a ) );
112 arg2 = sqrt( ( s-chord ) / ( 2.0*a ) );
113 Sinv = arg2;
114 Cosv = sqrt( 1.0 - (magro+magr-chord)/(4.0*a) );
115 BetE = 2.0*acos(Cosv);
116 BetE = 2.0*asin(Sinv);
117 if ( DNu > pi ):
118 BetE= -BetE;
119
120 Cosv= sqrt( 1.0 - s/(2.0*a) );
121 Sinv= arg1;
122 am = s*0.5;
123 ae = pi;
124 be = 2.0*asin( sqrt( (s-chord)/s ) );
125 tm = sqrt(am*am*am/mu)*(ae - (be-sin(be)));
126 if ( Dtsec > tm ):
127 AlpE= 2.0*pi-2.0*asin( Sinv );
128 else:
129 AlpE= 2.0*asin( Sinv );
130
131 DE = AlpE - BetE;
132 F = 1.0 - (a/magro)*(1.0 - cos(DE) );
133 GDot= 1.0 - (a/magr)* (1.0 - cos(DE) );
134 G = Dtsec - sqrt(a*a*a/mu)*(DE - sin(DE));
135 else:
136 # --------------------- Parabolic ---------------------
137 arg1 = 0.0;
138 arg2 = 0.0;
139 Exception(' a parabolic orbit ');
140
141 vo = v_create(3);
142 v = v_create(3);
143
144 for i in range(0, 3):
145 vo[i]= ( r[i] - F*ro[i] )/G;
146 v[i] = ( GDot*r[i] - ro[i] )/G;
147
148 return vo, v;
